def bfs(graph, start):
    visited, queue=set(), [start]
    while queue:
        vertex=queue.pop(0)
        if vertex not in visited:
            visited.add(vertex)
            queue.extend(graph[vertex]-visited)
    return visited
graph={
 'A':set(['B','C']),
 'B':set(['A','D','E']),
 'C':set(['A','F','G']),
 'D':set(['B']),
 'E':set(['B']),
 'F':set(['C']),
 'G':set(['C'])
}
bfs(graph, 'A')
